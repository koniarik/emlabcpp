45 template < typename T >
46 concept converter_check = requires()
47 {
48  {
49  T::max_size
50  } -> std::convertible_to< std::size_t >;
51  typename T::value_type;
52  requires bounded_derived< typename T::size_type >;
53 }
54 &&requires( std::span< uint8_t, T::max_size > buff, typename T::value_type item )
55 {
56  {
57  T::serialize_at( buff, item )
58  } -> std::same_as< typename T::size_type >;
59 }
60 &&requires( bounded_view< const uint8_t*, typename T::size_type > buff )
61 {
62  {
63  T::deserialize( buff )
64  } -> std::same_as< protocol_result< typename T::value_type > >;

68 struct converter< D, Endianess >
69 {
70  using value_type = typename proto_traits< D >::value_type;
71  static constexpr std::size_t max_size = proto_traits< D >::max_size;
72  using size_type = bounded< std::size_t, max_size, max_size >;
73 
74  static constexpr bool is_big_endian = Endianess == PROTOCOL_BIG_ENDIAN;
75 
76  static constexpr auto& bget( auto& buffer, std::size_t i )
77  {
78  return buffer[is_big_endian ? i : max_size - 1 - i];
79  }

81  static constexpr size_type
82  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )
83  {
84  serializer< value_type, Endianess >::serialize_at( buffer, item );
85  return size_type{};
86  }
87 
88  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
89  -> protocol_result< value_type >
90  {

96 template < convertible D, std::size_t N, endianess_enum Endianess >
97 struct converter< std::array< D, N >, Endianess >
98 {
99  using value_type = typename proto_traits< std::array< D, N > >::value_type;
100  static constexpr std::size_t max_size = proto_traits< std::array< D, N > >::max_size;
101  static constexpr std::size_t min_size = proto_traits< std::array< D, N > >::min_size;
102 
103  using sub_def = converter< D, Endianess >;
104  using sub_size_type = typename sub_def::size_type;
105  using size_type = bounded< std::size_t, min_size, max_size >;
106 
111 
112  static constexpr size_type
113  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
114  {
115  auto iter = buffer.begin();
116 
117  for ( std::size_t i : range( N ) ) {
118  std::span< uint8_t, sub_def::max_size > sub_view{ iter, sub_def::max_size };
119 
120  bounded bused = sub_def::serialize_at( sub_view, item[i] );
121 
122  std::advance( iter, *bused );
123  }
124 
125  auto used = std::distance( buffer.begin(), iter );
126  auto opt_bused = size_type::make( used );
127  EMLABCPP_ASSERT( opt_bused );
128  return *opt_bused;
129  }
130 
131  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
132  -> protocol_result< value_type >
133  {
134  value_type res{};
135  std::size_t offset = 0;
136 
137  for ( std::size_t i : range( N ) ) {
138  auto opt_view = buffer.template opt_offset< sub_size_type >( offset );
139 
140  if ( !opt_view ) {
141  return { offset, &SIZE_ERR };
142  }
143 
144  auto [used, subres] = sub_def::deserialize( *opt_view );
145  offset += used;
146 
147  if ( std::holds_alternative< const protocol_mark* >( subres ) ) {
148  return { offset, *std::get_if< const protocol_mark* >( &subres ) };
149  } else {
150  res[i] = *std::get_if< 0 >( &subres );
151  }
152  }
153 
154  return protocol_result{ offset, res };
155  }
156 };

158 template < convertible... Ds, endianess_enum Endianess >
159 struct converter< std::tuple< Ds... >, Endianess >
160 {
161  using def_type = std::tuple< Ds... >;
162 
163  using value_type = typename proto_traits< def_type >::value_type;
164  static constexpr std::size_t max_size = proto_traits< def_type >::max_size;
165  static constexpr std::size_t min_size =
166  ( converter< Ds, Endianess >::size_type::min_val + ... + 0 );
167  using size_type = bounded< std::size_t, min_size, max_size >;
168 
169  static constexpr size_type
170  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
171  {
172  auto iter = buffer.begin();
173 
174  for_each_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
175  using sub_def =
176  converter< std::tuple_element_t< i, def_type >, Endianess >;
177 
178  std::span< uint8_t, sub_def::max_size > sub_view{ iter, sub_def::max_size };
179 
180  bounded bused = sub_def::serialize_at( sub_view, std::get< i >( item ) );
181 
182  std::advance( iter, *bused );
183  } );
184 
185  auto used = std::distance( buffer.begin(), iter );
186  auto opt_bused = size_type::make( used );
187  EMLABCPP_ASSERT( opt_bused );
188  return *opt_bused;
189  }
190 
191  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
192  -> protocol_result< value_type >
193  {
194  value_type res;
195 
196  std::size_t offset = 0;
197  std::optional< const protocol_mark* > opt_err;
198 
199  until_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
200  using D = std::tuple_element_t< i, def_type >;
201  using sub_def = converter< D, Endianess >;
202 
203  auto opt_view =
204  buffer.template opt_offset< typename sub_def::size_type >( offset );
205 
206  if ( !opt_view ) {
207  opt_err = &SIZE_ERR;
208  return true;
209  }
210 
211  auto [sused, sres] = sub_def::deserialize( *opt_view );
212  offset += sused;
213  if ( std::holds_alternative< const protocol_mark* >( sres ) ) {
214  opt_err = *std::get_if< const protocol_mark* >( &sres );
215  return true;
216  } else {
217  std::get< i >( res ) = *std::get_if< 0 >( &sres );
218  }
219  return false;
220  } );
221 
222  if ( opt_err ) {
223  return { offset, *opt_err };
224  }
225  return { offset, res };
226  }

229 template < convertible... Ds, endianess_enum Endianess >
230 struct converter< std::variant< Ds... >, Endianess >
231 {
232  using def_type = std::variant< Ds... >;
233  using value_type = typename proto_traits< def_type >::value_type;
234  static constexpr std::size_t max_size = proto_traits< def_type >::max_size;
235 
236  using id_type = uint8_t;
237  using id_def = converter< id_type, Endianess >;
238  static constexpr std::size_t id_size = id_def::max_size;
239 
240  static_assert(
241  sizeof...( Ds ) < std::numeric_limits< id_type >::max(),
242  "Number of items for variant is limited by the size of one byte - 256 items" );
243  static_assert( protocol_fixedly_sized< id_type > );
244 
245  static constexpr std::size_t min_size =
246  id_def::size_type::min_val +
247  std::min( { converter< Ds, Endianess >::size_type::min_val... } );
248  using size_type = bounded< std::size_t, min_size, max_size >;
249 
250  static constexpr size_type
251  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
252  {
253  id_def::serialize_at(
254  buffer.template first< id_size >(), static_cast< id_type >( item.index() ) );
255 
256  std::optional< size_type > opt_res;
257  until_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
258  if ( i != item.index() ) {
259  return false;
260  }
261  using sub_def =
262  converter< std::variant_alternative_t< i, def_type >, Endianess >;
263 
265  opt_res =
266  bounded_constant< id_def::max_size > +
267  sub_def::serialize_at(
268  buffer.template subspan< id_def::max_size, sub_def::max_size >(),
269  std::get< i >( item ) );
270 
271  return true;
272  } );
275  EMLABCPP_ASSERT( opt_res );
276  return *opt_res;
277  }
278 
279  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
280  -> protocol_result< value_type >
281  {
282  protocol_result< value_type > res{ 0, &UNDEFVAR_ERR };
283 
284  auto [iused, idres] = id_def::deserialize( buffer.template first< id_size >() );
285  std::size_t used = iused;
286 
287  if ( std::holds_alternative< const protocol_mark* >( idres ) ) {
288  res.res = *std::get_if< const protocol_mark* >( &idres );
289  return res;
290  }
291  id_type id = std::get< 0 >( idres );
292 
293  auto item_view = buffer.template offset< id_size >();
294 
295  until_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
296  using D = std::variant_alternative_t< i, def_type >;
297  using sub_def = converter< D, Endianess >;
298 
299  if ( id != i ) {
300  return false;
301  }
302 
303  auto opt_view =
304  item_view.template opt_offset< typename sub_def::size_type >( 0 );
305 
306  if ( !opt_view ) {
307  res.res = &SIZE_ERR;
308  return true;
309  }
310 
311  auto [sused, sres] = sub_def::deserialize( *opt_view );
312  res.used = used + sused;
313  if ( std::holds_alternative< const protocol_mark* >( sres ) ) {
314  res.res = *std::get_if< const protocol_mark* >( &sres );
315  } else {
316  res.res = value_type{ *std::get_if< 0 >( &sres ) };
317  }
318  return true;
319  } );
320 
321  return res;
322  }
323 };

344 template < convertible T, endianess_enum Endianess >
345 struct converter< std::optional< T >, Endianess >
346 {
347  using decl = proto_traits< std::optional< T > >;
348  using value_type = std::optional< T >;
349  static constexpr std::size_t max_size = decl::max_size;
350  static constexpr std::size_t min_size = decl::min_size;
351  using presence_type = typename decl::presence_type;
352 
353  static_assert( protocol_fixedly_sized< presence_type > );
354 
355  using presence_def = converter< presence_type, Endianess >;
356  static constexpr std::size_t presence_size = presence_def::max_size;
357  using sub_def = converter< T, Endianess >;
358  using size_type = bounded< std::size_t, min_size, max_size >;
359 
360  static constexpr size_type
361  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& opt_val )
362  {
363  if ( !opt_val ) {
364  return presence_def::serialize_at(
365  buffer.template first< presence_size >(), 0 );
366  }
367  auto psize =
368  presence_def::serialize_at( buffer.template first< presence_size >(), 1 );
369 
370  return psize + sub_def::serialize_at(
371  buffer.template subspan< presence_size, sub_def::max_size >(),
372  *opt_val );
373  }
374 
375  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
376  -> protocol_result< value_type >
377  {
378  protocol_result< value_type > res{ 0, &BADVAL_ERR };
379 
380  auto [pused, pres] =
381  presence_def::deserialize( buffer.template first< presence_size >() );
382  res.used = pused;
383 
384  if ( std::holds_alternative< const protocol_mark* >( pres ) ) {
385  const protocol_mark* m = *std::get_if< const protocol_mark* >( &pres );
386  res.res = m;
387  return res;
388  }
389  auto is_present = *std::get_if< presence_type >( &pres );
390  if ( is_present == 0 ) {
391  res.res = std::nullopt;
392  return res;
393  }
394  if ( is_present != 1 ) {
395  return res;
396  }
397 
398  using sub_size = typename sub_def::size_type;
399  auto opt_view = buffer.template opt_offset< sub_size >( presence_size );
400 
401  if ( !opt_view ) {
402  return res;
403  }
404 
405  auto [sused, sres] = sub_def::deserialize( *opt_view );
406 
407  res.used += sused;
408  match(
409  sres,
410  [&]( T val ) {
411  res.res = std::make_optional( std::move( val ) );
412  },
413  [&]( const protocol_mark* m ) {
414  res.res = m;
415  } );
416  return res;
417  }
418 };

420 template < std::size_t N, endianess_enum Endianess >
421 struct converter< std::bitset< N >, Endianess >
422 {
423  using value_type = typename proto_traits< std::bitset< N > >::value_type;
424  static constexpr std::size_t max_size = proto_traits< std::bitset< N > >::max_size;
425  using size_type = bounded< std::size_t, max_size, max_size >;
426 
427  static constexpr bool is_big_endian = Endianess == PROTOCOL_BIG_ENDIAN;
428 
429  static constexpr auto& bget( auto& buffer, std::size_t i )
430  {
431  return buffer[is_big_endian ? i : max_size - 1 - i];
432  }
433 
434  static constexpr size_type
435  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
436  {
437  for ( std::size_t i : range( max_size ) ) {
438  std::bitset< 8 > byte;
439  for ( std::size_t j : range( 8u ) ) {
440  byte[j] = item[i * 8 + j];
441  }
442  bget( buffer, i ) = static_cast< uint8_t >( byte.to_ulong() );
443  }
444  return size_type{};
445  }
446 
447  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
448  -> protocol_result< value_type >
449  {
450  std::bitset< N > res;
451  for ( std::size_t i : range( max_size ) ) {
452  std::bitset< 8 > byte = bget( buffer, i );
453  for ( std::size_t j : range( 8u ) ) {
454  res[i * 8 + j] = byte[j];
455  }
456  }
457  return { max_size, res };
458  }

582 template < convertible CounterDef, convertible D, endianess_enum Endianess >
583 struct converter< protocol_sized_buffer< CounterDef, D >, Endianess >
584 {
585  using value_type =
586  typename proto_traits< protocol_sized_buffer< CounterDef, D > >::value_type;
587  static constexpr std::size_t max_size =
588  proto_traits< protocol_sized_buffer< CounterDef, D > >::max_size;
589 
590  using sub_def = converter< D, Endianess >;
591 
592  using counter_def = converter< CounterDef, Endianess >;
593  using counter_size_type = typename counter_def::size_type;
594  using counter_type = typename counter_def::value_type;
595  static constexpr std::size_t counter_size = counter_def::max_size;
596 
598  static_assert( protocol_fixedly_sized< CounterDef > );
599 
600  static constexpr std::size_t min_size = counter_size + sub_def::size_type::min_val;
601 
602  using size_type = bounded< std::size_t, min_size, max_size >;
603 
604  static constexpr size_type
605  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
606  {
607  auto vused =
608  sub_def::serialize_at( buffer.template last< sub_def::max_size >(), item );
609  counter_def::serialize_at(
610  buffer.template first< counter_size >(),
611  static_cast< counter_type >( *vused ) );
612  return vused + counter_size_type{};
613  }
614 
615  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
616  -> protocol_result< value_type >
617  {
618 
619  auto [cused, cres] =
620  counter_def::deserialize( buffer.template first< counter_size >() );
621 
622  if ( std::holds_alternative< const protocol_mark* >( cres ) ) {
623  return { cused, *std::get_if< const protocol_mark* >( &cres ) };
624  }
625  std::size_t used = *std::get_if< 0 >( &cres );
626  auto start_iter = buffer.begin() + counter_size;
627 
628  auto opt_view = bounded_view< const uint8_t*, typename sub_def::size_type >::make(
629  view_n( start_iter, used ) );

914  }

923  return { cused, *std::get_if< const protocol_mark* >( &cres ) };
924  }

933  buffer.template opt_offset< typename sub_def::size_type >( offset );

944  } else {

946  }
