61 template < typename T >
62 concept converter_check = requires()
63 {
64  {
65  T::max_size
66  } -> std::convertible_to< std::size_t >;
67  typename T::value_type;
68  requires bounded_derived< typename T::size_type >;

70 &&requires( std::span< uint8_t, T::max_size > buff, typename T::value_type item )

75 }
76 &&requires( bounded_view< const uint8_t*, typename T::size_type > buff )

91 struct converter< D, Endianess >
92 {
93  using value_type = typename traits_for< D >::value_type;
94  static constexpr std::size_t max_size = traits_for< D >::max_size;
95  using size_type = bounded< std::size_t, max_size, max_size >;
96 
97  static constexpr bool is_big_endian = Endianess == std::endian::big;
98 
99  static constexpr auto& bget( auto& buffer, const std::size_t i )
100  {

102  }

105  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )
106  {
107  serializer< value_type, Endianess >::serialize_at( buffer, item );

118 template < convertible D, std::size_t N, std::endian Endianess >
119 struct converter< std::array< D, N >, Endianess >
120 {
121  using value_type = typename traits_for< std::array< D, N > >::value_type;
122  static constexpr std::size_t max_size = traits_for< std::array< D, N > >::max_size;
123  static constexpr std::size_t min_size = traits_for< std::array< D, N > >::min_size;
124 
125  using sub_converter = converter_for< D, Endianess >;
126  using sub_size_type = typename sub_converter::size_type;
127  using size_type = bounded< std::size_t, min_size, max_size >;
128 
133 
134  static constexpr size_type
135  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
136  {
137  auto iter = buffer.begin();
138 
139  for ( const std::size_t i : range( N ) ) {
140  std::span< uint8_t, sub_converter::max_size > sub_view{
141  iter, sub_converter::max_size };
142 
143  bounded bused = sub_converter::serialize_at( sub_view, item[i] );
144 
145  std::advance( iter, *bused );
146  }
147 
148  auto used = std::distance( buffer.begin(), iter );
149  auto opt_bused = size_type::make( used );
150  EMLABCPP_ASSERT( opt_bused );
151  return *opt_bused;
152  }
153 
154  static constexpr error_possibility_enum error_posib =
155  ( fixedly_sized< D > && !erroring_converter< sub_converter > ) ? ERROR_IMPOSSIBLE :
156  ERROR_POSSIBLE;
157 
158  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
159  -> conversion_result< value_type, error_posib >
160  {
161 
162  value_type res{};
163  std::size_t offset = 0;
164 
165  for ( const std::size_t i : range( N ) ) {
166  auto opt_view = buffer.template opt_offset< sub_size_type >( offset );
167 
168  if constexpr ( error_posib == ERROR_POSSIBLE ) {
169  if ( !opt_view ) {
170  return { offset, &SIZE_ERR };
171  }
172  }
173 
174  auto sres = sub_converter::deserialize( *opt_view );
175  offset += sres.used;
176  if constexpr ( erroring_converter< sub_converter > ) {
177  if ( sres.has_error() ) {
178  return { offset, sres.get_error() };
179  }
180  }
181 
182  res[i] = *sres.get_value();
183  }
184 
185  return { offset, res };
186  }
187 };

189 template < convertible... Ds, std::endian Endianess >
190 struct converter< std::tuple< Ds... >, Endianess >
191 {
192  using def_type = std::tuple< Ds... >;
193 
194  using value_type = typename traits_for< def_type >::value_type;
195  static constexpr std::size_t max_size = traits_for< def_type >::max_size;
196  static constexpr std::size_t min_size = traits_for< def_type >::min_size;
197  using size_type = bounded< std::size_t, min_size, max_size >;
198 
199  static constexpr size_type
200  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
201  {
202  auto iter = buffer.begin();
203 
204  for_each_index< sizeof...( Ds ) >( [&iter, &item]< std::size_t i >() {
205  using sub_converter =
206  converter_for< std::tuple_element_t< i, def_type >, Endianess >;
207 
208  std::span< uint8_t, sub_converter::max_size > sub_view{
209  iter, sub_converter::max_size };
210 
211  bounded bused =
212  sub_converter::serialize_at( sub_view, std::get< i >( item ) );
213 
214  std::advance( iter, *bused );
215  } );
216 
217  auto used = std::distance( buffer.begin(), iter );
218  auto opt_bused = size_type::make( used );
219  EMLABCPP_ASSERT( opt_bused );
220  return *opt_bused;
221  }
222 
223  static constexpr error_possibility_enum error_posib =
224  fixedly_sized< def_type > &&
225  ( !erroring_converter< converter< Ds, Endianess > > && ... && true ) ?
226  ERROR_IMPOSSIBLE :
227  ERROR_POSSIBLE;
228 
229  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
230  -> conversion_result< value_type, error_posib >
231  {
232  value_type res;
233 
234  std::size_t offset = 0;
235  std::optional< const mark* > opt_err;
236 
237  until_index< sizeof...( Ds ) >(
238  [&offset, &opt_err, &res, &buffer]< std::size_t i >() {
239  using D = std::tuple_element_t< i, def_type >;
240  using sub_converter = converter_for< D, Endianess >;
241 
242  auto opt_view =
243  buffer.template opt_offset< typename sub_converter::size_type >(
244  offset );
245 
246  if constexpr ( !fixedly_sized< def_type > ) {
247  if ( !opt_view ) {
248  opt_err = &SIZE_ERR;
249  return true;
250  }
251  }
252 
253  auto sres = sub_converter::deserialize( *opt_view );
254  offset += sres.used;
255  if constexpr ( erroring_converter< sub_converter > ) {
256  if ( sres.has_error() ) {
257  opt_err = sres.get_error();
258  return true;
259  }
260  }
261 
262  std::get< i >( res ) = *sres.get_value();
263 
264  return false;
265  } );
266 
267  if constexpr ( error_posib == ERROR_POSSIBLE ) {
268  if ( opt_err ) {
269  return { offset, *opt_err };
270  }
271  }
272  return { offset, res };
273  }

277 template < convertible... Ds, std::endian Endianess >
278 struct converter< std::variant< Ds... >, Endianess >
279 {
280  using def_type = std::variant< Ds... >;
281  using value_type = typename traits_for< def_type >::value_type;
282  static constexpr std::size_t max_size = traits_for< def_type >::max_size;
283  static constexpr std::size_t min_size = traits_for< def_type >::min_size;
284 
285  using id_type = uint8_t;
286  using id_def = converter_for< id_type, Endianess >;
287  static constexpr std::size_t id_size = id_def::max_size;
288 
289  static_assert(
290  sizeof...( Ds ) < std::numeric_limits< id_type >::max(),
291  "Number of items for variant is limited by the size of one byte - 256 items" );
292  static_assert( fixedly_sized< id_type > );
293  using size_type = bounded< std::size_t, min_size, max_size >;
294 
295  static constexpr size_type
296  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
297  {
298  id_def::serialize_at(
299  buffer.template first< id_size >(), static_cast< id_type >( item.index() ) );
300 
301  return visit_index(
302  [&buffer, &item]< std::size_t i >() -> size_type {
303  using sub_converter = converter_for<
304  std::variant_alternative_t< i, def_type >,
305  Endianess >;
306 
308  return bounded_constant< id_def::max_size > +
309  sub_converter::serialize_at(
310  buffer.template subspan<
311  id_def::max_size,
312  sub_converter::max_size >(),
313  std::get< i >( item ) );
314  },
315  item );
316  }
317 
318  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
319  -> conversion_result< value_type >
320  {
321  return id_def::deserialize( buffer.template first< id_size >() )
322  .bind_value( [&buffer]( const std::size_t iused, const id_type id ) {
323  auto item_view = buffer.template offset< id_size >();
324 
325  conversion_result< value_type > res{ 0, &UNDEFVAR_ERR };
326  until_index< sizeof...( Ds ) >(
327  [&res, &item_view, &id, &iused]< std::size_t i >() {
328  using D = std::variant_alternative_t< i, def_type >;
329  using sub_converter = converter_for< D, Endianess >;
330 
331  if ( id != i ) {
332  return false;
333  }
334 
335  auto opt_view = item_view.template opt_offset<
336  typename sub_converter::size_type >( 0 );
337 
338  if ( !opt_view ) {
339  res.res = &SIZE_ERR;
340  return true;
341  }
342 
343  res = conversion_result< value_type >{
344  sub_converter::deserialize( *opt_view )
345  .convert_value( []( auto item ) {
346  return value_type{
347  std::in_place_index< i >, item };
348  } ) };
349  res.used += iused;
350  return true;
351  } );
352 
353  return res;
354  } );
355  }
356 };

377 template < convertible T, std::endian Endianess >
378 struct converter< std::optional< T >, Endianess >
379 {
380  using decl = traits_for< std::optional< T > >;
381  using value_type = std::optional< T >;
382  static constexpr std::size_t max_size = decl::max_size;
383  static constexpr std::size_t min_size = decl::min_size;
384  using presence_type = typename decl::presence_type;
385 
386  static_assert( fixedly_sized< presence_type > );
387 
388  using presence_def = converter_for< presence_type, Endianess >;
389  static constexpr std::size_t presence_size = presence_def::max_size;
390  using sub_converter = converter_for< T, Endianess >;
391  using size_type = bounded< std::size_t, min_size, max_size >;
392 
393  static constexpr size_type
394  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& opt_val )
395  {
396  if ( !opt_val ) {
397  return presence_def::serialize_at(
398  buffer.template first< presence_size >(), 0 );
399  }
400  auto psize =
401  presence_def::serialize_at( buffer.template first< presence_size >(), 1 );
402 
403  return psize +
404  sub_converter::serialize_at(
405  buffer.template subspan< presence_size, sub_converter::max_size >(),
406  *opt_val );
407  }
408 
409  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
410  -> conversion_result< value_type >
411  {
412  return presence_def::deserialize( buffer.template first< presence_size >() )
413  .bind_value(
414  [&buffer]( std::size_t pused, presence_type is_present )
415  -> conversion_result< value_type > {
416  if ( is_present == 0 ) {
417  return { pused, value_type{} };
418  }
419  if ( is_present != 1 ) {
420  return { pused, &BADVAL_ERR };
421  }
422 
423  using sub_size = typename sub_converter::size_type;
424  auto opt_view =
425  buffer.template opt_offset< sub_size >( presence_size );
426 
427  if ( !opt_view ) {
428  return { pused, &BADVAL_ERR };
429  }
430 
431  auto res = sub_converter::deserialize( *opt_view )
432  .convert_value( []( T item ) {
433  return std::make_optional( item );
434  } );
435  res.used += pused;
436  return conversion_result< value_type >{ std::move( res ) };
437  } );
438  }
439 };

441 template < std::size_t N, std::endian Endianess >
442 struct converter< std::bitset< N >, Endianess >
443 {
444  using value_type = typename traits_for< std::bitset< N > >::value_type;
445  static constexpr std::size_t max_size = traits_for< std::bitset< N > >::max_size;
446  using size_type = bounded< std::size_t, max_size, max_size >;
447 
448  static constexpr bool is_big_endian = Endianess == std::endian::big;
449 
450  static constexpr auto& bget( auto& buffer, const std::size_t i )
451  {
452  return buffer[is_big_endian ? i : max_size - 1 - i];
453  }
454 
455  static constexpr size_type
456  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
457  {
458  for ( const std::size_t i : range( max_size ) ) {
459  std::bitset< 8 > byte;
460  for ( const std::size_t j : range( 8u ) ) {
461  byte[j] = item[i * 8 + j];
462  }
463  bget( buffer, i ) = static_cast< uint8_t >( byte.to_ulong() );
464  }
465  return size_type{};
466  }
467 
468  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
469  -> conversion_result< value_type >
470  {
471  std::bitset< N > res;
472  for ( const std::size_t i : range( max_size ) ) {
473  std::bitset< 8 > byte = bget( buffer, i );
474  for ( const std::size_t j : range( 8u ) ) {
475  res[i * 8 + j] = byte[j];
476  }
477  }
478  return { max_size, res };
479  }
