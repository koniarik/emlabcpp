60 template < typename T >
61 concept converter_check = requires()
62 {
63  {
64  T::max_size
65  } -> std::convertible_to< std::size_t >;
66  typename T::value_type;
67  requires bounded_derived< typename T::size_type >;
68 }
69 &&requires( std::span< uint8_t, T::max_size > buff, typename T::value_type item )
70 {
71  {
72  T::serialize_at( buff, item )
73  } -> std::same_as< typename T::size_type >;
74 }
75 &&requires( std::span< const uint8_t > buff, typename T::value_type item )
76 {
77  T::deserialize( buff, item );

80 template < base_type D, std::endian Endianess >
81 struct converter< D, Endianess >
82 {
83  using value_type = typename traits_for< D >::value_type;
84  static constexpr std::size_t max_size = traits_for< D >::max_size;
85  using size_type = bounded< std::size_t, max_size, max_size >;
86 
87  static constexpr bool is_big_endian = Endianess == std::endian::big;
88 
89  static constexpr auto& bget( auto& buffer, const std::size_t i )
90  {
91  return buffer[is_big_endian ? i : max_size - 1 - i];
92  }
93 
94  static constexpr size_type
95  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )
96  {
97  serializer< value_type, Endianess >::serialize_at( buffer, item );
98  return size_type{};
99  }
100 
101  static constexpr conversion_result
102  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
103  {
104  if ( buffer.size() < max_size ) {
105  return { 0, &SIZE_ERR };
106  }
107 
108  value =
109  serializer< value_type, Endianess >::deserialize( buffer.first< max_size >() );
110 
111  return conversion_result{ max_size };
112  }

115 template < convertible D, std::size_t N, std::endian Endianess >
116 struct converter< std::array< D, N >, Endianess >
117 {
118  using value_type = typename traits_for< std::array< D, N > >::value_type;
119  static constexpr std::size_t max_size = traits_for< std::array< D, N > >::max_size;
120  static constexpr std::size_t min_size = traits_for< std::array< D, N > >::min_size;
121 
122  using sub_converter = converter_for< D, Endianess >;
123  using sub_size_type = typename sub_converter::size_type;
124  using size_type = bounded< std::size_t, min_size, max_size >;
125 
130 
131  static constexpr size_type
132  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
133  {
134  auto iter = buffer.begin();
135 
136  for ( const std::size_t i : range( N ) ) {
137  std::span< uint8_t, sub_converter::max_size > sub_view{
138  iter, sub_converter::max_size };
139 
140  bounded bused = sub_converter::serialize_at( sub_view, item[i] );
141 
142  std::advance( iter, *bused );
143  }
144 
145  auto used = std::distance( buffer.begin(), iter );
146  auto opt_bused = size_type::make( used );
147  EMLABCPP_ASSERT( opt_bused );
148  return *opt_bused;
149  }
150 
151  static constexpr conversion_result
152  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
153  {
154 
155  std::size_t offset = 0;
156 
157  for ( const std::size_t i : range( N ) ) {
158  if ( offset > buffer.size() ) {
159  return { offset, &SIZE_ERR };
160  }
161  std::span subspan = buffer.subspan( offset );
162 
163  auto sres = sub_converter::deserialize( subspan, value[i] );
164  if ( sres.has_error() ) {
165  return sres;
166  }
167  offset += sres.used;
168  }
169 
170  return conversion_result{ offset };
171  }
172 };

174 template < convertible... Ds, std::endian Endianess >
175 struct converter< std::tuple< Ds... >, Endianess >
176 {
177  using def_type = std::tuple< Ds... >;
178 
179  using value_type = typename traits_for< def_type >::value_type;
180  static constexpr std::size_t max_size = traits_for< def_type >::max_size;
181  static constexpr std::size_t min_size = traits_for< def_type >::min_size;
182  using size_type = bounded< std::size_t, min_size, max_size >;
183 
184  static constexpr size_type
185  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
186  {
187  auto iter = buffer.begin();
188 
189  for_each_index< sizeof...( Ds ) >( [&iter, &item]< std::size_t i >() {
190  using sub_converter =
191  converter_for< std::tuple_element_t< i, def_type >, Endianess >;
192 
193  std::span< uint8_t, sub_converter::max_size > sub_view{
194  iter, sub_converter::max_size };
195 
196  bounded bused =
197  sub_converter::serialize_at( sub_view, std::get< i >( item ) );
198 
199  std::advance( iter, *bused );
200  } );
201 
202  auto used = std::distance( buffer.begin(), iter );
203  auto opt_bused = size_type::make( used );
204  EMLABCPP_ASSERT( opt_bused );
205  return *opt_bused;
206  }
207 
208  template < std::size_t I >
209  using nth_converter = converter_for< std::tuple_element_t< I, def_type >, Endianess >;
210 
211  static constexpr conversion_result
212  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
213  {
214 
215  std::size_t offset = 0;
216  const mark* err = nullptr;
217 
218  until_index< sizeof...( Ds ) >(
219  [&offset, &err, &value, &buffer]< std::size_t i >() {
220  if ( offset > buffer.size() ) {
221  err = &SIZE_ERR;
222  return true;
223  }
224  std::span subspan = buffer.subspan( offset );
225 
226  auto sres =
227  nth_converter< i >::deserialize( subspan, std::get< i >( value ) );
228 
229  offset += sres.used;
230 
231  if ( sres.has_error() ) {
232  err = sres.get_error();
233  return true;
234  }
235  return false;
236  } );
237 
238  return conversion_result{ offset, err };
239  }

244 struct converter< std::variant< Ds... >, Endianess >
245 {
246  using def_type = std::variant< Ds... >;
247  using value_type = typename traits_for< def_type >::value_type;
248  static constexpr std::size_t max_size = traits_for< def_type >::max_size;
249  static constexpr std::size_t min_size = traits_for< def_type >::min_size;
250 
251  using id_type = uint8_t;
252  using id_converter = converter_for< id_type, Endianess >;
253  static constexpr std::size_t id_size = id_converter::max_size;
254 
255  static_assert(
256  sizeof...( Ds ) < std::numeric_limits< id_type >::max(),
257  "Number of items for variant is limited by the size of one byte - 256 items" );
258  static_assert( fixedly_sized< id_type > );
259  using size_type = bounded< std::size_t, min_size, max_size >;
260 
261  static constexpr size_type
262  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
263  {
264  id_converter::serialize_at(
265  buffer.template first< id_size >(), static_cast< id_type >( item.index() ) );
266 
267  return visit_index(
268  [&buffer, &item]< std::size_t i >() -> size_type {
269  using sub_converter = converter_for<
270  std::variant_alternative_t< i, def_type >,
271  Endianess >;
272 
274  return bounded_constant< id_converter::max_size > +
275  sub_converter::serialize_at(
276  buffer.template subspan<
277  id_converter::max_size,
278  sub_converter::max_size >(),
279  std::get< i >( item ) );
280  },
281  item );
282  }
283 
284  template < std::size_t I >
285  using nth_converter = converter_for< std::variant_alternative_t< I, def_type >, Endianess >;
286 
287  static constexpr conversion_result
288  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
289  {
290  id_type id;
291  auto subres = id_converter::deserialize( buffer, id );
292  if ( subres.has_error() ) {
293  return subres;
294  }
295 
296  auto subspan = buffer.subspan< id_size >();
297 
298  conversion_result res{ 0, &UNDEFVAR_ERR };
299 
300  until_index< sizeof...( Ds ) >(
301  [&res, &subres, &subspan, &id, &value]< std::size_t i >() {
302  if ( id != i ) {
303  return false;
304  }
305 
306  res = nth_converter< i >::deserialize(
307  subspan, value.template emplace< i >() );
308  res.used += subres.used;
309  return true;
310  } );
311 
312  return res;
313  }

335 template < convertible T, std::endian Endianess >
336 struct converter< std::optional< T >, Endianess >
337 {
338  using traits = traits_for< std::optional< T > >;
339  using value_type = std::optional< T >;
340  static constexpr std::size_t max_size = traits::max_size;
341  static constexpr std::size_t min_size = traits::min_size;
342  using presence_type = typename traits::presence_type;
343 
344  static_assert( fixedly_sized< presence_type > );
345 
346  using presence_converter = converter_for< presence_type, Endianess >;
347  static constexpr std::size_t presence_size = presence_converter::max_size;
348  using sub_converter = converter_for< T, Endianess >;
349  using sub_size = typename sub_converter::size_type;
350  using size_type = bounded< std::size_t, min_size, max_size >;
351 
352  static constexpr presence_type is_present = bounded< uint8_t, 0, 1 >::get< 1 >();
353  static constexpr presence_type not_present = bounded< uint8_t, 0, 1 >::get< 0 >();
354 
355  static constexpr size_type
356  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& opt_val )
357  {
358  if ( !opt_val ) {
359  return presence_converter::serialize_at(
360  buffer.template first< presence_size >(), not_present );
361  }
362  auto psize = presence_converter::serialize_at(
363  buffer.template first< presence_size >(), is_present );
364 
365  return psize +
366  sub_converter::serialize_at(
367  buffer.template subspan< presence_size, sub_converter::max_size >(),
368  *opt_val );
369  }
370 
371  static constexpr conversion_result
372  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
373  {
374  presence_type is_present_v;
375  auto subres = presence_converter::deserialize(
376  buffer.first< presence_size >(), is_present_v );
377  if ( subres.has_error() ) {
378  return subres;
379  }
380 
381  if ( is_present_v == not_present ) {
382  return subres;
383  }
384 
385  std::span subspan = buffer.subspan( presence_size );
386 
387  auto res = sub_converter::deserialize( subspan, value.emplace() );
388  res.used += subres.used;
389  return res;
390  }
391 };

393 template < std::size_t N, std::endian Endianess >
394 struct converter< std::bitset< N >, Endianess >
395 {
396  using value_type = typename traits_for< std::bitset< N > >::value_type;
397  static constexpr std::size_t max_size = traits_for< std::bitset< N > >::max_size;
398  using size_type = bounded< std::size_t, max_size, max_size >;
399 
400  static constexpr bool is_big_endian = Endianess == std::endian::big;
401 
402  static constexpr auto& bget( auto& buffer, const std::size_t i )
403  {
404  return buffer[is_big_endian ? i : max_size - 1 - i];
405  }
406 
407  static constexpr size_type
408  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
409  {
410  for ( const std::size_t i : range( max_size ) ) {
411  std::bitset< 8 > byte;
412  for ( const std::size_t j : range( 8u ) ) {
413  byte[j] = item[i * 8 + j];
414  }
415  bget( buffer, i ) = static_cast< uint8_t >( byte.to_ulong() );
416  }
417  return size_type{};
418  }
419 
420  static constexpr conversion_result
421  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
422  {
423  if ( buffer.size() < max_size ) {
424  return { 0, &SIZE_ERR };
425  }
426  for ( const std::size_t i : range( max_size ) ) {
427  std::bitset< 8 > byte = bget( buffer, i );
428  for ( const std::size_t j : range( 8u ) ) {
429  value[i * 8 + j] = byte[j];
430  }
431  }
432  return conversion_result{ max_size };
433  }
434 };

437 struct converter< message< N >, Endianess >
438 {
439  using traits_type = traits_for< message< N > >;
440  using value_type = typename traits_type::value_type;
441  static constexpr std::size_t min_size = traits_type::min_size;
442  static constexpr std::size_t max_size = traits_type::max_size;
443  using size_type = bounded< std::size_t, min_size, max_size >;
444  using msg_size_type = typename traits_type::msg_size_type;
445  static_assert( fixedly_sized< msg_size_type > );
446  using msg_size_converter = converter_for< msg_size_type, Endianess >;
447  static constexpr std::size_t msg_size_size = msg_size_converter::max_size;
448 
449  static constexpr size_type
450  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
451  {
452  auto used = msg_size_converter::serialize_at(
453  buffer.template first< msg_size_size >(),
454  static_cast< msg_size_type >( item.size() ) );
455 
456  for ( const std::size_t i : range( item.size() ) ) {
457  buffer[i + msg_size_size] = item[i];
458  }
460  auto opt_bused = size_type::make( item.size() + *used );
461  EMLABCPP_ASSERT( opt_bused );
462  return *opt_bused;
463  }
464 
465  static constexpr conversion_result
466  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
467  {
468  msg_size_type size;
469  auto subres = msg_size_converter::deserialize( buffer, size );
470  if ( subres.has_error() ) {
471  return subres;
472  }
473  if ( buffer.size() < subres.used + size ) {
474  return { subres.used, &SIZE_ERR };
475  }
476  auto opt_msg = value_type::make( buffer.subspan( subres.used, size ) );
477 
478  if ( !opt_msg ) {
479  return { msg_size_size, &BIGSIZE_ERR };
480  }
481  value = *opt_msg;
482  return conversion_result{ opt_msg->size() + subres.used };
483  }

606 template < convertible CounterDef, convertible D, std::endian Endianess >
607 struct converter< sized_buffer< CounterDef, D >, Endianess >
608 {
609  using value_type = typename traits_for< sized_buffer< CounterDef, D > >::value_type;
610  static constexpr std::size_t max_size =
611  traits_for< sized_buffer< CounterDef, D > >::max_size;
612  static constexpr std::size_t min_size =
613  traits_for< sized_buffer< CounterDef, D > >::min_size;
614 
615  using sub_converter = converter_for< D, Endianess >;
616 
617  using counter_converter = converter_for< CounterDef, Endianess >;
618  using counter_size_type = typename counter_converter::size_type;
619  using counter_type = typename counter_converter::value_type;
620  static constexpr std::size_t counter_size = counter_converter::max_size;
621 
623  static_assert( fixedly_sized< CounterDef > );
624 
625  using size_type = bounded< std::size_t, min_size, max_size >;
626 
627  static constexpr size_type
628  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
629  {
630  auto vused = sub_converter::serialize_at(
631  buffer.template last< sub_converter::max_size >(), item );
632  auto cused = counter_converter::serialize_at(
633  buffer.template first< counter_size >(),
634  static_cast< counter_type >( *vused ) );
635  return vused + cused;
636  }

643  if ( cres.has_error() ) {
644  return cres;
645  }
646  if ( buffer.size() < cres.used + size ) {

870 struct converter< static_vector< T, N >, Endianess >
871 {
872  using value_type = typename traits_for< static_vector< T, N > >::value_type;
873 
874  static_assert( N <= std::numeric_limits< uint16_t >::max() );
875 
876  using counter_type = typename traits_for< static_vector< T, N > >::counter_type;
877  using counter_converter = converter_for< counter_type, Endianess >;
878  using sub_converter = converter_for< T, Endianess >;
879 
880  static constexpr std::size_t counter_size = counter_converter::max_size;
881 
882  static_assert( fixedly_sized< counter_type > );
883 
884  static constexpr std::size_t max_size = traits_for< static_vector< T, N > >::max_size;
885  static constexpr std::size_t min_size = traits_for< static_vector< T, N > >::min_size;
886  using size_type = bounded< std::size_t, min_size, max_size >;
887 
888  static constexpr size_type
889  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )
890  {
891 
892  counter_converter::serialize_at(
893  buffer.template first< counter_size >(),
894  static_cast< counter_type >( item.size() ) );
895 
897  auto iter = buffer.begin() + counter_size;
898  for ( const std::size_t i : range( item.size() ) ) {
899  std::span< uint8_t, sub_converter::max_size > sub_view{
900  iter, sub_converter::max_size };
901 
902  bounded sub_bused = sub_converter::serialize_at( sub_view, item[i] );
903 
904  std::advance( iter, *sub_bused );
905  }
906 
907  auto used = static_cast< std::size_t >( std::distance( buffer.begin(), iter ) );
908  auto opt_bused = size_type::make( used );
909  EMLABCPP_ASSERT( opt_bused );
910 
911  return *opt_bused;
912  }
913 
914  static constexpr conversion_result
915  deserialize( const std::span< const uint8_t >& buffer, value_type& value )
916  {
917  counter_type count;
918  auto subres = counter_converter::deserialize( buffer, count );
919  if ( subres.has_error() ) {
920  return subres;
921  }
922  std::size_t offset = subres.used;
923 
924  for ( const std::size_t i : range( count ) ) {
925  std::ignore = i;
926  if ( offset > buffer.size() ) {
927  return { offset, &SIZE_ERR };
928  }
929  std::span subspan = buffer.subspan( offset );
930  value.emplace_back();
931 
932  auto itemres = sub_converter::deserialize( subspan, value.back() );
933  offset += itemres.used;
934  if ( itemres.has_error() ) {
935  return { offset, itemres.get_error() };
936  }
937  }
938 
939  return conversion_result{ offset };
940  }
