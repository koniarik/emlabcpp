constexpr auto decompose( T&& item ) \
{ \
if constexpr ( std::is_rvalue_reference_v< T > ) { \
auto&& [__VA_ARGS__] = std::move( item ); \
return std::make_tuple( __VA_ARGS__ ); \
} else { \

} \
