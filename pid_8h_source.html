102 template < typename TimeType >
103 float update( pid< TimeType >& pid, TimeType now, float measured, float desired )
104 {
105  if ( now == pid.last_time ) {
106  return pid.output;
107  }
108 
109  auto t_diff = static_cast< float >( now - pid.last_time );
110 
111  const pid_coefficients& coeff = pid.cfg.coefficients;
112 
113  const float error = desired - measured;
114  pid.i_sum += coeff.i * ( error * t_diff );
115  pid.i_sum = std::clamp( pid.i_sum, pid.cfg.limits.min, pid.cfg.limits.max );
116 
117  const float measured_diff = ( measured - pid.last_measured ) / t_diff;
118  pid.output = coeff.p * error + pid.i_sum - coeff.d * measured_diff;
119  pid.output = std::clamp( pid.output, pid.cfg.limits.min, pid.cfg.limits.max );
120 
121  pid.last_measured = measured;
122  pid.last_time = now;
123 
124  return pid.output;
125 }
