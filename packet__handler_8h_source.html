49  static message_type serialize( const value_type& val )
50  {
51 
52  message_type msg = sub_handler::serialize(
53  std::make_tuple( Packet::prefix, size_type{}, val, checksum_type{} ) );
54 
55  auto size = static_cast< size_type >(
56  msg.size() - Packet::prefix_decl::max_size - size_size );
57 
58  serializer< size_type, endianess >::serialize_at(
59  std::span< uint8_t, size_size >{ msg.begin() + size_offset, size_size }, size );
60 
61  checksum_type chcksm =
62  Packet::get_checksum( view_n( msg.begin(), msg.size() - checksum_size ) );
63 
64  serializer< checksum_type, endianess >::serialize_at(
65  std::span< uint8_t, checksum_size >{ msg.end() - checksum_size, checksum_size },
66  chcksm );
67 
68  return msg;
69  }
70 
71  static either< value_type, error_record >
72  extract( const view< const uint8_t* >& msg )
73  {
74  return sub_handler::extract( msg ).bind_left(
75  [&]( std::tuple< prefix_type, size_type, value_type, checksum_type > pack )
76  -> either< value_type, error_record > {
77  checksum_type present_checksum = std::get< 3 >( pack );
78  std::size_t checksum_pos = msg.size() - checksum_size;
79  checksum_type calculated_checksum =
80  Packet::get_checksum( view_n( msg.begin(), checksum_pos ) );
81  if ( present_checksum != calculated_checksum ) {
82  EMLABCPP_LOG(
83  "Problematic message: " << *message_type::make( msg ) );
84  EMLABCPP_LOG(
85  std::hex << "Checksum failed, calculated: "
86  << int( calculated_checksum )
87  << " present: " << int( present_checksum ) );
88  return error_record{ CHECKSUM_ERR, checksum_pos };
89  }
90  return std::get< 2 >( pack );
91  } );
92  }
