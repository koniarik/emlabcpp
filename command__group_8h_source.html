113  template < auto id, typename... Args >

115  make_val( const Args&... args )
116  {
117  std::optional< value_type > res;

120  using cmd = std::tuple_element_t< i, cmds_type >;
121  if constexpr ( cmd::id == id ) {
122  res.emplace( cmd::make_val( args... ) );
123  }
124  } );
