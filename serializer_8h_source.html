35 template < protocol_base_type T, protocol_endianess_enum Endianess >
36 struct protocol_serializer
37 {
38  static constexpr std::size_t max_size = sizeof( T );
39  using size_type = bounded< std::size_t, max_size, max_size >;
40  using view_type = bounded_view< const uint8_t*, size_type >;
41  static constexpr bool is_big_endian = Endianess == PROTOCOL_BIG_ENDIAN;
42 
43  static constexpr auto& bget( auto& buffer, std::size_t i )
44  {
45  return buffer[is_big_endian ? i : max_size - 1 - i];
46  }
47  static constexpr void serialize_at( std::span< uint8_t, max_size > buffer, T item )
48  {
49  for ( std::size_t i : range( max_size ) ) {
50  bget( buffer, max_size - i - 1 ) = static_cast< uint8_t >( item & 0xFF );
51  item = static_cast< T >( item >> 8 );
52  }
53  }
54  static constexpr T deserialize( const view_type& buffer )
55  {
56  T res{};
57  for ( std::size_t i : range( max_size ) ) {
58  res = static_cast< T >( res << 8 );
59  res = static_cast< T >( res | bget( buffer, i ) );
60  }
61  return { res };
62  }
63 };

65 template < protocol_endianess_enum Endianess >
66 struct protocol_serializer< float, Endianess >
67 {
68  using sub_serializer = protocol_serializer< uint32_t, Endianess >;
69  static_assert( sizeof( float ) == sizeof( uint32_t ) );
70 
71  static constexpr std::size_t max_size = sizeof( float );
72  using size_type = bounded< std::size_t, max_size, max_size >;
73  using view_type = bounded_view< const uint8_t*, size_type >;
74 
75  static constexpr void serialize_at( std::span< uint8_t, max_size > buffer, float item )
76  {
77  uint32_t v;
78  std::memcpy( &v, &item, sizeof( float ) );
79  sub_serializer::serialize_at( buffer, v );
80  }
81 
82  static constexpr float deserialize( const view_type& buffer )
83  {
84  uint32_t v = sub_serializer::deserialize( buffer );
85  float res;
86  std::memcpy( &res, &v, sizeof( float ) );
87  return res;
88  }
