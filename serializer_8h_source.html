34 template < protocol_base_type T, protocol_endianess_enum Endianess >
35 struct protocol_serializer
36 {
37  static constexpr std::size_t max_size = sizeof( T );
38  using size_type = bounded< std::size_t, max_size, max_size >;
39  using view_type = bounded_view< const uint8_t*, size_type >;
40  static constexpr bool is_big_endian = Endianess == PROTOCOL_BIG_ENDIAN;
41 
42  static constexpr auto& bget( auto& buffer, std::size_t i )
43  {
44  return buffer[is_big_endian ? i : max_size - 1 - i];
45  }
46  static constexpr void serialize_at( std::span< uint8_t, max_size > buffer, T item )
47  {
48  for ( std::size_t i : range( max_size ) ) {
49  bget( buffer, max_size - i - 1 ) = static_cast< uint8_t >( item & 0xFF );
50  item = static_cast< T >( item >> 8 );
51  }
52  }
53  static constexpr T deserialize( const view_type& buffer )
54  {
55  T res{};
56  for ( std::size_t i : range( max_size ) ) {
57  res = static_cast< T >( res << 8 );
58  res = static_cast< T >( res | bget( buffer, i ) );
59  }
60  return { res };
61  }
62 };
