35 template < base_type T, std::endian Endianess >
36 struct serializer
37 {
38  static constexpr std::size_t max_size = sizeof( T );
39  using size_type = bounded< std::size_t, max_size, max_size >;
40  static constexpr bool is_big_endian = Endianess == std::endian::big;
41 
42  static constexpr auto& bget( auto& buffer, const std::size_t i )
43  {
44  return buffer[is_big_endian ? i : max_size - 1 - i];
45  }
46  static constexpr void serialize_at( std::span< uint8_t, max_size > buffer, T item )
47  {
48  for ( const std::size_t i : range( max_size ) ) {
49  bget( buffer, max_size - i - 1 ) = static_cast< uint8_t >( item & 0xFF );
50  item = static_cast< T >( item >> 8 );
51  }
52  }
53  static constexpr T deserialize( const std::span< const uint8_t, max_size >& buffer )
54  {
55  T res{};
56  for ( std::size_t i : range( max_size ) ) {
57  res = static_cast< T >( res << 8 );
58  res = static_cast< T >( res | bget( buffer, i ) );
59  }
60  return { res };
61  }
62 };

65 struct serializer< float, Endianess >
66 {
67  using sub_serializer = serializer< uint32_t, Endianess >;
68  static_assert( sizeof( float ) == sizeof( uint32_t ) );
69 
70  static constexpr std::size_t max_size = sizeof( float );
71  using size_type = bounded< std::size_t, max_size, max_size >;
72 
73  static constexpr void
74  serialize_at( std::span< uint8_t, max_size > buffer, const float item )
75  {
76  uint32_t v = 0;
77  std::memcpy( &v, &item, sizeof( float ) );
78  sub_serializer::serialize_at( buffer, v );

82  {
83  const uint32_t v = sub_serializer::deserialize( buffer );
