158 inline std::optional< ser_res >
159 ser_cell( uint32_t key, std::span< std::byte > value, std::span< std::byte, cell_size > dest )
160 {
161  if ( value.empty() || key > sin_bit_mask )
162  return {};
163 
164  static_assert( sizeof( uint32_t ) == hcell_size );
165  if ( value.size() <= hcell_size ) {
166  uint32_t val = 0x00;
167  std::memcpy( &val, value.data(), value.size() );
168  uint64_t tmp = key | sin_bit_mask;
169  tmp = ( tmp << 32 ) + val;
170 
171  std::memcpy( dest.data(), &tmp, cell_size );
172  return ser_res::SINGLE;
173  } else {

175  closest_multiple_of( static_cast< uint32_t >( value.size() ), cell_size ) /
176  cell_size;
177  uint64_t tmp = key;
178  tmp = ( tmp << 32 ) + size;

192 {
193  uint64_t tmp = 0x00;

195  auto front = static_cast< uint32_t >( tmp >> 32 );
196  deser_res r{

198  .key = static_cast< uint32_t >( front & key_mask ),
199  .val = static_cast< uint32_t >( tmp & 0xFFFF'FFFF ),
200  };
201 
202  if ( r.is_seq && r.val == 0 )
