116 constexpr quaternion slerp( const quaternion& q, const quaternion& s, float f )
117 {
118  // NOTE: inspired by tf::Quaternion::slerp
119  const float theta = angle_shortest_path( q, s ) / 2.0f;
120  if ( theta == 0.0f )
121  return q;
122 
123  const float d = 1.0f / std::sin( theta );
124  const float s0 = std::sin( ( 1.0f - f ) * theta );
125  const float s1 = std::sin( f * theta );
126  float m = 1.0f;
127  if ( dot( q, s ) < 0 )
128  m = -1.0f;
129  return {
130  ( q[0] * s0 + m * s[0] * s1 ) * d,
131  ( q[1] * s0 + m * s[1] * s1 ) * d,
132  ( q[2] * s0 + m * s[2] * s1 ) * d,
133  ( q[3] * s0 + m * s[3] * s1 ) * d,
134  };
135 }
