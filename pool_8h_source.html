49 template < std::size_t PoolSize, uint16_t PoolCount >
50 class pool_resource final : public pool_interface
51 {
52 public:
53  pool_resource()
54  {
55  for ( std::size_t i : range( PoolCount ) ) {
56  free_.push_back( static_cast< uint16_t >( i ) );
57  }
58  }
59 
60  pool_resource( const pool_resource& ) = delete;
61  pool_resource( pool_resource&& ) = delete;
62  pool_resource& operator=( const pool_resource& ) = delete;
63  pool_resource& operator=( pool_resource&& ) = delete;
64 
65  void* allocate( std::size_t bytes, std::size_t alignment ) final
66  {
67  void* p = nullptr;
68  std::size_t used = bytes;
69  if ( !free_.empty() ) {
70  std::size_t i = free_.back();
71  void* pool_p = &pools_[i];
72  p = align( pool_p, alignment );
73  used += static_cast< std::size_t >(
74  reinterpret_cast< std::byte* >( p ) -
75  reinterpret_cast< std::byte* >( pool_p ) );
76  }
77 
78  if ( p == nullptr || used > PoolSize ) {
79 #ifdef __EXCEPTIONS
80  throw std::bad_alloc{};
81 #else
82  EMLABCPP_ASSERT( false );
83  return nullptr;
84 #endif
85  }
86  free_.pop_back();
87  return p;
88  }
89 
90  void deallocate( void* ptr, std::size_t ) final
91  {
92 
93  auto pval = reinterpret_cast< std::size_t >( ptr );
94  auto bval = reinterpret_cast< std::size_t >( &pools_ );
95 
96  std::size_t spot_i = ( pval - bval ) / PoolSize;
97 
98  if ( spot_i >= PoolCount ) {
99 #ifdef __EXCEPTIONS
100  throw std::bad_alloc{};
101 #else
102  EMLABCPP_ASSERT( false );
103  return;
104 #endif
105  }
106  free_.push_back( static_cast< uint16_t >( spot_i ) );
107  }
108 
109  bool full() final
110  {
111  return free_.empty();
112  }
113 
114  ~pool_resource() final = default;
115 
116 private:
117  using pool = std::array< std::byte, PoolSize >;
118 
119  static_vector< uint16_t, PoolCount > free_;
120  std::array< pool, PoolCount > pools_;
121 };
