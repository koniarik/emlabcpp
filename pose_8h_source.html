126 lineary_interpolate_path( const std::vector< pose >& ipath, float d_step, float a_step )
127 {
128  std::vector< pose > res;
129  if ( ipath.empty() ) {
130  return res;
131  }
132  for ( std::size_t i : range( ipath.size() - 1 ) ) {
133  const pose& from = ipath[i];
134  const pose& to = ipath[i + 1];
135  std::size_t seg_steps = steps( distance_of( to, from ), d_step, a_step );
136  for ( std::size_t j : range( seg_steps ) ) {
137  res.push_back( lin_interp( from, to, float( j ) / float( seg_steps ) ) );
138  }
