703 template < typename T, std::size_t N, protocol_endianess_enum Endianess >
704 struct protocol_def< static_vector< T, N >, Endianess >
705 {
706  using value_type = typename protocol_decl< static_vector< T, N > >::value_type;
707 
708  static_assert( N <= std::numeric_limits< uint16_t >::max() );
709 
710  using counter_type = typename protocol_decl< static_vector< T, N > >::counter_type;
711  using counter_def = protocol_def< counter_type, Endianess >;
712  using sub_def = protocol_def< T, Endianess >;
713 
714  static constexpr std::size_t counter_size = counter_def::max_size;
715 
716  static_assert( protocol_fixedly_sized< counter_type > );
717 
718  static constexpr std::size_t max_size = protocol_decl< static_vector< T, N > >::max_size;
719  static constexpr std::size_t min_size = counter_size;
720  using size_type = bounded< std::size_t, min_size, max_size >;
721 
722  static constexpr size_type
723  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )
724  {
725 
726  counter_def::serialize_at(
727  buffer.template first< counter_size >(),
728  static_cast< counter_type >( item.size() ) );
729 
731  auto iter = buffer.begin() + counter_size;
732  for ( std::size_t i : range( item.size() ) ) {
733  std::span< uint8_t, sub_def::max_size > sub_view{ iter, sub_def::max_size };
734 
735  bounded sub_bused = sub_def::serialize_at( sub_view, item[i] );
736 
737  std::advance( iter, *sub_bused );
738  }
739 
740  auto used = static_cast< std::size_t >( std::distance( buffer.begin(), iter ) );
741  auto opt_bused = size_type::make( used );
742  EMLABCPP_ASSERT( opt_bused );
743 
744  return *opt_bused;
745  }
746 
747  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
748  -> protocol_result< value_type >
749  {
750  auto [cused, cres] =
751  counter_def::deserialize( buffer.template first< counter_size >() );
752 
753  if ( std::holds_alternative< const protocol_mark* >( cres ) ) {
754  return { cused, *std::get_if< const protocol_mark* >( &cres ) };
755  }
756  std::size_t cnt = *std::get_if< 0 >( &cres );
757  std::size_t offset = counter_size;
758  value_type res{};
759 
760  for ( std::size_t i : range( cnt ) ) {
761  std::ignore = i;
762 
763  auto opt_view =
764  buffer.template opt_offset< typename sub_def::size_type >( offset );
765  if ( !opt_view ) {
766  return { offset, &SIZE_ERR };
767  }
768 
769  auto [used, subres] = sub_def::deserialize( *opt_view );
770 
771  offset += used;
772 
773  if ( std::holds_alternative< const protocol_mark* >( subres ) ) {
774  return { offset, *std::get_if< const protocol_mark* >( &subres ) };
775  } else {
776  res.push_back( *std::get_if< 0 >( &subres ) );
777  }
778  }
779 
780  return { offset, res };
781  }
782 };
