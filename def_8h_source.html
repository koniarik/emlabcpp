45 template < typename T >
46 concept protocol_def_check = requires()
47 {
48  {
49  T::max_size
50  } -> std::convertible_to< std::size_t >;
51  typename T::value_type;
52  bounded_derived< typename T::size_type >;
53 }
54 &&requires( std::span< uint8_t, T::max_size > buff, typename T::value_type item )
55 {
56  {
57  T::serialize_at( buff, item )
58  } -> std::same_as< typename T::size_type >;
59 }
60 &&requires( bounded_view< const uint8_t*, typename T::size_type > buff )
61 {
62  {
63  T::deserialize( buff )
64  } -> std::same_as< protocol_result< typename T::value_type > >;

68 struct protocol_def< D, Endianess >
69 {

72  using size_type = bounded< std::size_t, max_size, max_size >;

75

82  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )

96 template < protocol_declarable D, std::size_t N, protocol_endianess_enum Endianess >
97 struct protocol_def< std::array< D, N >, Endianess >
98 {
99  using value_type = typename protocol_decl< std::array< D, N > >::value_type;
100  static constexpr std::size_t max_size = protocol_decl< std::array< D, N > >::max_size;
101 
102  using sub_def = protocol_def< D, Endianess >;
103  using sub_size_type = typename sub_def::size_type;
104  using size_type = bounded< std::size_t, sub_def::size_type::min_val * N, max_size >;
105 
110 
111  static constexpr size_type
112  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
113  {
114  auto iter = buffer.begin();
115 
116  for ( std::size_t i : range( N ) ) {
117  std::span< uint8_t, sub_def::max_size > sub_view{ iter, sub_def::max_size };
118 
119  bounded bused = sub_def::serialize_at( sub_view, item[i] );
120 
121  std::advance( iter, *bused );
122  }
123 
124  auto used = std::distance( buffer.begin(), iter );
125  auto opt_bused = size_type::make( used );
126  EMLABCPP_ASSERT( opt_bused );
127  return *opt_bused;
128  }
129 
130  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
131  -> protocol_result< value_type >
132  {
133  value_type res{};
134  std::size_t offset = 0;
135 
136  for ( std::size_t i : range( N ) ) {
137  auto opt_view = buffer.template opt_offset< sub_size_type >( offset );
138 
139  if ( !opt_view ) {
140  return { offset, &SIZE_ERR };
141  }
142 
143  auto [used, subres] = sub_def::deserialize( *opt_view );
144  offset += used;
145 
146  if ( std::holds_alternative< const protocol_mark* >( subres ) ) {
147  return { offset, *std::get_if< const protocol_mark* >( &subres ) };
148  } else {
149  res[i] = *std::get_if< 0 >( &subres );
150  }
151  }
152 
153  return protocol_result{ offset, res };
154  }
155 };

157 template < protocol_declarable... Ds, protocol_endianess_enum Endianess >
158 struct protocol_def< std::tuple< Ds... >, Endianess >
159 {
160  using def_type = std::tuple< Ds... >;
161 
162  using value_type = typename protocol_decl< def_type >::value_type;
163  static constexpr std::size_t max_size = protocol_decl< def_type >::max_size;
164  static constexpr std::size_t min_size =
165  ( protocol_def< Ds, Endianess >::size_type::min_val + ... + 0 );
166  using size_type = bounded< std::size_t, min_size, max_size >;
167 
168  static constexpr size_type
169  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
170  {
171  auto iter = buffer.begin();
172 
173  for_each_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
174  using sub_def =
175  protocol_def< std::tuple_element_t< i, def_type >, Endianess >;
176 
177  std::span< uint8_t, sub_def::max_size > sub_view{ iter, sub_def::max_size };
178 
179  bounded bused = sub_def::serialize_at( sub_view, std::get< i >( item ) );
180 
181  std::advance( iter, *bused );
182  } );
183 
184  auto used = std::distance( buffer.begin(), iter );
185  auto opt_bused = size_type::make( used );
186  EMLABCPP_ASSERT( opt_bused );
187  return *opt_bused;
188  }
189 
190  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
191  -> protocol_result< value_type >
192  {
193  value_type res;
194 
195  std::size_t offset = 0;
196  std::optional< const protocol_mark* > opt_err;
197 
198  until_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
199  using D = std::tuple_element_t< i, def_type >;
200  using sub_def = protocol_def< D, Endianess >;
201 
202  auto opt_view =
203  buffer.template opt_offset< typename sub_def::size_type >( offset );
204 
205  if ( !opt_view ) {
206  opt_err = &SIZE_ERR;
207  return true;
208  }
209 
210  auto [sused, sres] = sub_def::deserialize( *opt_view );
211  offset += sused;
212  if ( std::holds_alternative< const protocol_mark* >( sres ) ) {
213  opt_err = *std::get_if< const protocol_mark* >( &sres );
214  return true;
215  } else {
216  std::get< i >( res ) = *std::get_if< 0 >( &sres );
217  }
218  return false;
219  } );
220 
221  if ( opt_err ) {
222  return { offset, *opt_err };
223  }
224  return { offset, res };
225  }

228 template < protocol_declarable... Ds, protocol_endianess_enum Endianess >
229 struct protocol_def< std::variant< Ds... >, Endianess >
230 {
231  using def_type = std::variant< Ds... >;
232  using value_type = typename protocol_decl< def_type >::value_type;
233  static constexpr std::size_t max_size = protocol_decl< def_type >::max_size;
234 
235  using id_type = uint8_t;
236  using id_def = protocol_def< id_type, Endianess >;
237  static constexpr std::size_t id_size = id_def::max_size;
238 
239  static_assert(
240  sizeof...( Ds ) < std::numeric_limits< id_type >::max(),
241  "Number of items for variant is limited by the size of one byte - 256 items" );
242  static_assert( protocol_fixedly_sized< id_type > );
243 
244  static constexpr std::size_t min_size =
245  id_def::size_type::min_val +
246  std::min( { protocol_def< Ds, Endianess >::size_type::min_val... } );
247  using size_type = bounded< std::size_t, min_size, max_size >;
248 
249  static constexpr size_type
250  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
251  {
252  id_def::serialize_at(
253  buffer.template first< id_size >(), static_cast< id_type >( item.index() ) );
254 
255  std::optional< size_type > opt_res;
256  until_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
257  if ( i != item.index() ) {
258  return false;
259  }
260  using sub_def =
261  protocol_def< std::variant_alternative_t< i, def_type >, Endianess >;
262 
264  opt_res =
265  bounded_constant< id_def::max_size > +
266  sub_def::serialize_at(
267  buffer.template subspan< id_def::max_size, sub_def::max_size >(),
268  std::get< i >( item ) );
269 
270  return true;
271  } );
274  EMLABCPP_ASSERT( opt_res );
275  return *opt_res;
276  }
277 
278  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
279  -> protocol_result< value_type >
280  {
281  protocol_result< value_type > res{ 0, &UNDEFVAR_ERR };
282 
283  auto [iused, idres] = id_def::deserialize( buffer.template first< id_size >() );
284  std::size_t used = iused;
285 
286  if ( std::holds_alternative< const protocol_mark* >( idres ) ) {
287  res.res = *std::get_if< const protocol_mark* >( &idres );
288  return res;
289  }
290  id_type id = std::get< 0 >( idres );
291 
292  auto item_view = buffer.template offset< id_size >();
293 
294  until_index< sizeof...( Ds ) >( [&]< std::size_t i >() {
295  using D = std::variant_alternative_t< i, def_type >;
296  using sub_def = protocol_def< D, Endianess >;
297 
298  if ( id != i ) {
299  return false;
300  }
301 
302  auto opt_view =
303  item_view.template opt_offset< typename sub_def::size_type >( 0 );
304 
305  if ( !opt_view ) {
306  res.res = &SIZE_ERR;
307  return true;
308  }
309 
310  auto [sused, sres] = sub_def::deserialize( *opt_view );
311  res.used = used + sused;
312  if ( std::holds_alternative< const protocol_mark* >( sres ) ) {
313  res.res = *std::get_if< const protocol_mark* >( &sres );
314  } else {
315  res.res = value_type{ *std::get_if< 0 >( &sres ) };
316  }
317  return true;
318  } );
319 
320  return res;
321  }
322 };

343 template < protocol_declarable T, protocol_endianess_enum Endianess >
344 struct protocol_def< std::optional< T >, Endianess >
345 {
346  using decl = protocol_decl< std::optional< T > >;
347  using value_type = std::optional< T >;
348  static constexpr std::size_t max_size = decl::max_size;
349  static constexpr std::size_t min_size = decl::min_size;
350  using presence_type = typename decl::presence_type;
351 
352  static_assert( protocol_fixedly_sized< presence_type > );
353 
354  using presence_def = protocol_def< presence_type, Endianess >;
355  static constexpr std::size_t presence_size = presence_def::max_size;
356  using sub_def = protocol_def< T, Endianess >;
357  using size_type = bounded< std::size_t, min_size, max_size >;
358 
359  static constexpr size_type
360  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& opt_val )
361  {
362  if ( !opt_val ) {
363  return presence_def::serialize_at(
364  buffer.template first< presence_size >(), 0 );
365  }
366  auto psize =
367  presence_def::serialize_at( buffer.template first< presence_size >(), 1 );
368 
369  return psize + sub_def::serialize_at(
370  buffer.template subspan< presence_size, sub_def::max_size >(),
371  *opt_val );
372  }
373 
374  static constexpr auto deserialize( bounded_view< const uint8_t*, size_type > buffer )
375  -> protocol_result< value_type >
376  {
377  protocol_result< value_type > res{ 0, &BADVAL_ERR };
378 
379  auto [pused, pres] =
380  presence_def::deserialize( buffer.template first< presence_size >() );
381  res.used = pused;
382 
383  if ( std::holds_alternative< const protocol_mark* >( pres ) ) {
384  const protocol_mark* m = *std::get_if< const protocol_mark* >( &pres );
385  res.res = m;
386  return res;
387  }
388  auto is_present = *std::get_if< presence_type >( &pres );
389  if ( is_present == 0 ) {
390  res.res = std::nullopt;
391  return res;
392  }
393  if ( is_present != 1 ) {
394  return res;
395  }
396 
397  using sub_size = typename sub_def::size_type;
398  auto opt_view = buffer.template opt_offset< sub_size >( presence_size );
399 
400  if ( !opt_view ) {
401  return res;
402  }
403 
404  auto [sused, sres] = sub_def::deserialize( *opt_view );
405 
406  res.used += sused;
407  match(
408  sres,
409  [&]( T val ) {
410  res.res = std::make_optional( std::move( val ) );
411  },
412  [&]( const protocol_mark* m ) {
413  res.res = m;
414  } );
415  return res;
416  }
417 };

420 struct protocol_def< std::bitset< N >, Endianess >
421 {
422  using value_type = typename protocol_decl< std::bitset< N > >::value_type;
423  static constexpr std::size_t max_size = protocol_decl< std::bitset< N > >::max_size;
424  using size_type = bounded< std::size_t, max_size, max_size >;
425 
426  static constexpr bool is_big_endian = Endianess == PROTOCOL_BIG_ENDIAN;
427 
428  static constexpr auto& bget( auto& buffer, std::size_t i )
429  {
430  return buffer[is_big_endian ? i : max_size - 1 - i];
431  }
432 
433  static constexpr size_type
434  serialize_at( std::span< uint8_t, max_size > buffer, const value_type& item )
435  {
436  for ( std::size_t i : range( max_size ) ) {
437  std::bitset< 8 > byte;
438  for ( std::size_t j : range( 8u ) ) {
439  byte[j] = item[i * 8 + j];
440  }
441  bget( buffer, i ) = static_cast< uint8_t >( byte.to_ulong() );
442  }
443  return size_type{};
444  }
445 
446  static constexpr auto deserialize( const bounded_view< const uint8_t*, size_type >& buffer )
447  -> protocol_result< value_type >
448  {
449  std::bitset< N > res;
450  for ( std::size_t i : range( max_size ) ) {
451  std::bitset< 8 > byte = bget( buffer, i );
452  for ( std::size_t j : range( 8u ) ) {
453  res[i * 8 + j] = byte[j];
454  }
455  }
456  return { max_size, res };

822 struct protocol_def< static_vector< T, N >, Endianess >
823 {
824  using value_type = typename protocol_decl< static_vector< T, N > >::value_type;
825 
826  static_assert( N <= std::numeric_limits< uint16_t >::max() );
827 
828  using counter_type = typename protocol_decl< static_vector< T, N > >::counter_type;
829  using counter_def = protocol_def< counter_type, Endianess >;
830  using sub_def = protocol_def< T, Endianess >;
831 
832  static constexpr std::size_t counter_size = counter_def::max_size;
833 
834  static_assert( protocol_fixedly_sized< counter_type > );
835 
836  static constexpr std::size_t max_size = protocol_decl< static_vector< T, N > >::max_size;
837  static constexpr std::size_t min_size = counter_size;
838  using size_type = bounded< std::size_t, min_size, max_size >;
839 
840  static constexpr size_type
841  serialize_at( std::span< uint8_t, max_size > buffer, value_type item )
842  {
843 
844  counter_def::serialize_at(
845  buffer.template first< counter_size >(),
846  static_cast< counter_type >( item.size() ) );
847 
849  auto iter = buffer.begin() + counter_size;
850  for ( std::size_t i : range( item.size() ) ) {
851  std::span< uint8_t, sub_def::max_size > sub_view{ iter, sub_def::max_size };
852 
853  bounded sub_bused = sub_def::serialize_at( sub_view, item[i] );
854 
855  std::advance( iter, *sub_bused );
856  }
857 
858  auto used = static_cast< std::size_t >( std::distance( buffer.begin(), iter ) );
859  auto opt_bused = size_type::make( used );
860  EMLABCPP_ASSERT( opt_bused );
861 
862  return *opt_bused;
863  }
864

866  -> protocol_result< value_type >
867  {
868  auto [cused, cres] =
869  counter_def::deserialize( buffer.template first< counter_size >() );
870 
871  if ( std::holds_alternative< const protocol_mark* >( cres ) ) {
872  return { cused, *std::get_if< const protocol_mark* >( &cres ) };
873  }
874  std::size_t cnt = *std::get_if< 0 >( &cres );
875  std::size_t offset = counter_size;
876  value_type res{};
877 
878  for ( std::size_t i : range( cnt ) ) {
879  std::ignore = i;
880 
881  auto opt_view =
882  buffer.template opt_offset< typename sub_def::size_type >( offset );
883  if ( !opt_view ) {
884  return { offset, &SIZE_ERR };
885  }
886 
887  auto [used, subres] = sub_def::deserialize( *opt_view );
888 
889  offset += used;
890 
891  if ( std::holds_alternative< const protocol_mark* >( subres ) ) {
892  return { offset, *std::get_if< const protocol_mark* >( &subres ) };
893  } else {
894  res.push_back( *std::get_if< 0 >( &subres ) );
895  }
896  }
897 
898  return { offset, res };
899  }
