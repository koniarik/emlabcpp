58  template < int Exp, char... Unit >
59  static consteval auto get_single_unit()
60  {
61  static_assert( Exp > -10 && Exp < 10 );
62  constexpr std::size_t unit_n = sizeof...( Unit );
63 
64  if constexpr ( Exp == 0 ) {
65  return std::array< char, 0 >{};
66  } else if constexpr ( Exp == 1 ) {
67  return std::array< char, unit_n >{ Unit... };
68  } else {
69  constexpr std::size_t char_n = unit_n + 1 + ( Exp > 0 ? 1 : 2 );
70  std::array< char, char_n > buffer{ Unit... };
71  std::size_t n = sizeof...( Unit );
72  buffer[n++] = '^';
73  if ( Exp < 0 ) {
74  buffer[n++] = '-';
75  buffer[n++] = -Exp + '0';
76  } else {
77  buffer[n++] = Exp + '0';
78  }
79  return buffer;
80  }
81  }
