60  static std::string get_unit()
61  {
62  auto seg = []( std::string unit, int i ) -> std::string {
63  if ( i == 0 ) {
64  return "";
65  }
66  if ( i == 1 ) {
67  return unit;
68  }
69  return unit + "^" + std::to_string( i );
70  };
71  return seg( "m", Len ) + seg( "g", Mass ) + seg( "s", Time ) + seg( "A", Current ) +
72  seg( "K", Temp ) + seg( "mol", Mol ) + seg( "cd", Li ) +
73  seg( "rad", Angle ) + seg( "B", Byte );
74  }
