62 inline opt< ser_res >
63 ser_cell( uint32_t key, std::span< std::byte > value, std::span< std::byte, cell_size > dest )
64 {
65  if ( value.empty() || key > sin_bit_mask )
66  return {};
67 
68  static_assert( sizeof( uint32_t ) == hcell_size );
69  if ( value.size() <= hcell_size ) {
70  uint32_t val = 0x00;
71  std::memcpy( &val, value.data(), value.size() );
72  uint64_t tmp = key | sin_bit_mask;
73  tmp = ( tmp << 32 ) + val;
74 
75  std::memcpy( dest.data(), &tmp, cell_size );
76  return ser_res::SINGLE;
77  } else {
78  uint32_t size =
79  closest_multiple_of( static_cast< uint32_t >( value.size() ), cell_size ) /
80  cell_size;
81  uint64_t tmp = key;
82  tmp = ( tmp << 32 ) + size;
83  std::memcpy( dest.data(), &tmp, cell_size );
84  return ser_res::MULTI;
85  }

96 {
