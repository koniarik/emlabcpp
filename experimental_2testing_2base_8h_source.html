101  testing_data_node* find_node( testing_node_id nid )
102  {
103  if ( id == nid ) {
104  return this;
105  }
106  auto iter = find_if( children, [&]( testing_data_node& child ) {
107  return child.find_node( nid );
108  } );
109  if ( iter == children.end() ) {
110  return nullptr;
111  }
112  return &*iter;
