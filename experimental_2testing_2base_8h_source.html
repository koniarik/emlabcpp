121 {

124  switch ( j.type() ) {

154 inline std::optional< testing_tree >
155 json_to_testing_tree( pool_interface* mem_pool, const nlohmann::json& inpt )
156 {
157  testing_tree tree{ mem_pool };
158 
159  static_function< testing_node_id( const nlohmann::json& j ), 32 > f =
160  [&]( const nlohmann::json& j ) -> testing_node_id {
161  if ( j.is_object() ) {
162  std::optional opt_res = tree.make_object_node();
163  if ( !opt_res ) {
164  // TODO: might wanna improve this
165  throw std::exception{};
166  }
167  auto [nid, oh] = *opt_res;
168  for ( const auto& [key, value] : j.items() ) {
169  testing_node_id chid = f( value );
170  oh.set( json_to_testing_key( key ), chid );
171  }
172  return nid;
173  }
174  if ( j.is_array() ) {
175  std::optional opt_res = tree.make_array_node();
176  if ( !opt_res ) {
177  // TODO: might wanna improve this
178  throw std::exception{};
179  }
180  auto [nid, ah] = *opt_res;
181  for ( const nlohmann::json& jj : j ) {
182  testing_node_id chid = f( jj );
183  ah.append( chid );
184  }
185  return nid;
186  }
187  std::optional opt_id = tree.make_value_node( json_to_testing_value( j ) );
188  if ( !opt_id ) {
189  throw std::exception{};
190  }
191  return *opt_id;
192  };
193 
194  try {
195  f( inpt );
196  return tree;
197  }
198  catch ( const std::exception& e ) {
199  EMLABCPP_LOG( "Build of testing tree failed because " << e.what() );
200  return std::nullopt;
201  }
202 }

204 inline nlohmann::json testing_tree_to_json( const testing_tree& tree )
205 {
206  static_function< nlohmann::json( testing_node_id ), 32 > f =
207  [&]( testing_node_id nid ) -> nlohmann::json {
208  const testing_node* node_ptr = tree.get_node( nid );
209 
210  if ( !node_ptr ) {
211  return {};
212  }
213 
214  return match(
215  node_ptr->get_container_handle(),
216  [&]( const testing_value& val ) {
217  return testing_value_to_json( val );
218  },
219  [&]( testing_const_object_handle oh ) {
220  nlohmann::json j;
221  // TODO: write iterators for handle
222  for ( testing_child_id chid : range( oh.size() ) ) {
223  const testing_key* k = oh.get_key( chid );
224  std::string key{ k->begin(), k->size() };
225  j[key] = f( *oh.get_child( chid ) );
226  }
227  return j;
228  },
229  [&]( testing_const_array_handle ah ) {
230  nlohmann::json j;
231  // TODO: write iterators for handle
232  for ( testing_child_id chid : range( ah.size() ) ) {
233  j.push_back( f( *ah.get_child( chid ) ) );
234  }
235  return j;
236  } );
237  };
238 
239  return f( 0 );
240 }
