143 inline std::optional< testing_tree >
144 json_to_testing_tree( pool_interface* mem_pool, const nlohmann::json& inpt )
145 {
146  testing_tree tree{ mem_pool };
147 
148  static_function< testing_node_id( const nlohmann::json& j ), 32 > f =
149  [&]( const nlohmann::json& j ) -> testing_node_id {
150  if ( j.is_object() ) {
151  std::optional opt_res = tree.make_object_node();
152  if ( !opt_res ) {
153  // TODO: might wanna improve this
154  throw std::exception{};
155  }
156  auto [nid, oh] = *opt_res;
157  for ( const auto& [key, value] : j.items() ) {
158  testing_node_id chid = f( value );
159  oh.set( json_to_testing_key( key ), chid );
160  }
161  return nid;
162  }
163  if ( j.is_array() ) {
164  std::optional opt_res = tree.make_array_node();
165  if ( !opt_res ) {
166  // TODO: might wanna improve this
167  throw std::exception{};
168  }
169  auto [nid, ah] = *opt_res;
170  for ( const nlohmann::json& jj : j ) {
171  testing_node_id chid = f( jj );
172  ah.append( chid );
173  }
174  return nid;
175  }
176  std::optional opt_id = tree.make_value_node( json_to_testing_value( j ) );
177  if ( !opt_id ) {
178  throw std::exception{};
179  }
180  return *opt_id;
181  };
182 
183  try {
184  f( inpt );
185  return tree;
186  }
187  catch ( std::exception& ) {
188  return std::nullopt;
189  }
190 }
