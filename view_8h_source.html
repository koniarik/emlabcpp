136 template < typename IteratorLh, typename IteratorRh >
137 constexpr bool operator==( const view< IteratorLh >& lh, const view< IteratorRh >& rh )
138 {
139  if ( lh.size() != rh.size() ) {
140  return false;
141  }
142 
143  IteratorLh lhiter = lh.begin();
144  IteratorRh rhiter = rh.begin();
145 
146  for ( ; lhiter != lh.end(); ++lhiter, ++rhiter ) {
