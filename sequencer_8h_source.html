59  either< sequencer_read_request, message_type > get_message()
60  {
61  auto bend = buffer_.end();
62  while ( !buffer_.empty() ) {
63  auto [piter, biter] =
64  std::mismatch( prefix.begin(), prefix.end(), buffer_.begin(), bend );
65 
67  if ( biter == buffer_.begin() ) {
68  buffer_.pop_front();
69  continue;
70  }
71 
73  if ( piter != prefix.end() && biter != bend ) {
74  buffer_.pop_front();
75  continue;
76  }
77 
79  if ( piter != prefix.end() ) {
80  return sequencer_read_request{
81  fixed_size - static_cast< std::size_t >(
82  std::distance( prefix.begin(), piter ) ) };
83  }
84 
85  break;
86  }
87 
88  if ( buffer_.empty() ) {
89  return sequencer_read_request{ fixed_size };
90  }
91 
92  std::size_t bsize = buffer_.size();
93 
96  EMLABCPP_ASSERT( bsize >= prefix.size() );
97 
98  if ( bsize < fixed_size ) {
99  return sequencer_read_request{ fixed_size - bsize };
100  }
101 
102  std::size_t desired_size = Def::get_size( buffer_ );
103  if ( bsize < desired_size ) {
104  return sequencer_read_request{ desired_size - bsize };
105  }
106 
107  auto opt_msg = message_type::make( view_n( buffer_.begin(), desired_size ) );
108  EMLABCPP_ASSERT( opt_msg );
109 
111  for ( std::size_t i = desired_size; i > 0; i-- ) {
112  buffer_.pop_front();
113  }
114 
115  return *opt_msg;
116  }
117 };

119 template < typename Sequencer, typename ReadCallback >
120 std::optional< typename Sequencer::message_type >
121 sequencer_simple_load( const std::size_t read_limit, ReadCallback&& read )
122 {
123  Sequencer seq;
124  std::optional< typename Sequencer::message_type > res;
125  std::size_t to_read = Sequencer::fixed_size;
126  std::size_t count = 0;
127  while ( !res && count < read_limit ) {
128  std::optional data = read( to_read );
129  if ( !data ) {
130  return res;
131  }
132  seq.insert( *data );
133  seq.get_message().match(
134  [&to_read, &count]( const std::size_t next_read ) {
135  to_read = next_read;
136  count = 0;
137  },
138  [&res]( auto msg ) {
139  res = msg;
140  } );
141 
142  count += 1;
143  }
144  return res;
145 }
