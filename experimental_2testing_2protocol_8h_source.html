224  static constexpr checksum_type get_checksum( const view< const uint8_t* > msg )
225  {
226  uint8_t init = 0x0;
227  return accumulate( msg, init, [&]( uint8_t accum, uint8_t val ) -> uint8_t {
228  return accum ^ val;
229  } );
230  }
