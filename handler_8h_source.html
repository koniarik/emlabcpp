37 template < convertible T >
38 struct handler
39 {
40  using def = converter_for< T, std::endian::big >;
41  using value_type = typename def::value_type;
42  using message_type = message< def::max_size >;
43 
44  static message_type serialize( const value_type& val )
45  {
46  std::array< uint8_t, def::max_size > buffer{};
47 
48  bounded used = def::serialize_at( buffer, val );
49  EMLABCPP_ASSERT( *used <= def::max_size );
50  return *message_type::make( view_n( buffer.begin(), *used ) );
51  };
52 
53  static either< value_type, error_record > extract( const view< const uint8_t* >& msg )
54  {
55  auto opt_view = bounded_view< const uint8_t*, typename def::size_type >::make(
56  view_n( msg.begin(), std::min( def::max_size, msg.size() ) ) );
57  if ( !opt_view.has_value() ) {
58  EMLABCPP_LOG(
59  "Failed to build view over provided message - wrong size: "
60  << msg.size() << " vs " << def::size_type::max_val );
61  return error_record{ SIZE_ERR, 0 };
62  }
63  value_type val;
64  auto res = def::deserialize( *opt_view, val );
65  if ( res.has_error() ) {
66  const mark* const mark = res.get_error();
67  EMLABCPP_LOG(
68  "Failed to extract protocol def "
69  << pretty_type_name< T >() << " from message " << *opt_view
70  << ", error is: " << *mark << " with " << res.used << " bytes" );
71  return error_record{ *mark, res.used };
72  }
73  return val;
74  }
75 };
